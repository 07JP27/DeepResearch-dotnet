// Alternative configuration examples:

// For Node.js MCP server:
var nodeConfig = new McpServerConfig
{
    Name = "Node Search Server",
    Command = "node",
    Arguments = ["search-server.js"]
};

// For .NET MCP server:
var dotnetConfig = new McpServerConfig
{
    Name = "DotNet Search Server", 
    Command = "dotnet",
    Arguments = ["run", "--project", "SearchServer"]
};

// For executable MCP server:
var executableConfig = new McpServerConfig
{
    Name = "Executable Search Server",
    Command = "./search-server",
    Arguments = ["--port", "8080"]
};
